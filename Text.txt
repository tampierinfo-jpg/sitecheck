  <script>
    function analyze(url) {
      let result = { url, status: "OK", reasons: [] };
      if (!url.startsWith("https")) { result.status = "Unsicher"; result.reasons.push("Kein HTTPS"); }
      if (url.includes("paypal") || url.includes("login")) { result.status = "Warnung"; result.reasons.push("VerdÃ¤chtiges Wort in URL"); }
      return result;
    }
    
    function checkUrl() {
      const url = document.getElementById("singleUrl").value;
      const res = analyze(url);
      showResult([res]);
    }
    
    function checkList() {
      const urls = document.getElementById("multiUrls").value.split("\n");
      const results = urls.filter(u => u.trim() !== "").map(u => analyze(u.trim()));
      showResult(results);
    }
    
    function showResult(results) {
      const container = document.getElementById("results");
      container.innerHTML = "";
      results.forEach(r => {
        const div = document.createElement("div");
        div.className = "result " + (r.status === "OK" ? "ok" : r.status === "Warnung" ? "warn" : "bad");
        div.innerHTML = `<b>${r.url}</b><br>Status: ${r.status}<br>${r.reasons.join(", ")}`;
        container.appendChild(div);
      });
    }
